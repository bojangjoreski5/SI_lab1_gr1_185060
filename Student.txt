class Student {
    String index;
    String firstName;
    String lastName;
    ArrayList labPoints;

    //TODO constructor

    public Student(String index, String firstName, String lastName, ArrayList labPoints) {
        this.index = index;
        this.firstName = firstName;
        this.lastName = lastName;
        ArrayList labPoints=new ArrayList(labPoints);
        for (int i=0;i<labPoints;i++){
            this.labPoints[i]=labPoints;
        }
    }


    //TODO seters & getters


    public String getIndex() {
        return index;
    }

    public String getFirstName() {
        return firstName;
    }

    public String getLastName() {
        return lastName;
    }

    public ArrayList getLabPoints() {
        return labPoints;
    }

    public void setIndex(String index) {
        this.index = index;
    }

    public void setFirstName(String firstName) {
        this.firstName = firstName;
    }

    public void setLastName(String lastName) {
        this.lastName = lastName;
    }

    public void setLabPoints(ArrayList labPoints) {
        this.labPoints = labPoints;
    }

    public double getAverage() {
        //TODO
        int vk=0;
        for (int i=0;i<labPoints;i++){
            vk+=labPoints[i];
        }
        return vk/labPoints;
    }

    public boolean hasSignature() {
        //TODO
        if (labPoints.size()>8){
            return true;
        }
        else {
            return false;
        }
    }
}
